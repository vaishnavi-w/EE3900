\section{Matrices 2.8}
If A = $\myvec{\frac{2}{3}&1&\frac{5}{3} \\ \frac{1}{3}&\frac{2}{3}&\frac{4}{3} \\ \frac{7}{3}&2&\frac{2}{3}}$ and B = $\myvec{\frac{2}{5}&\frac{3}{5}&1 \\ \frac{1}{5}&\frac{2}{5}&\frac{4}{5} \\ \frac{7}{5}&\frac{6}{5}&\frac{2}{5}}$, then compute 3A - 5B
\section{Solution}
\begin{align}
    3\vec{A} = 3\myvec{\frac{2}{3}&1&\frac{5}{3} \\ \frac{1}{3}&\frac{2}{3}&\frac{4}{3} \\ \frac{7}{3}&2&\frac{2}{3}} = \myvec{2&3&5\\ 1&2&4 \\ 7&6&2}\\
    5\vec{B} = 5\myvec{\frac{2}{5}&\frac{3}{5}&1 \\ \frac{1}{5}&\frac{2}{5}&\frac{4}{5} \\ \frac{7}{5}&\frac{6}{5}&\frac{2}{5}} = \myvec{2&3&5\\ 1&2&4 \\ 7&6&2}\\
    3\vec{A} - 5\vec{B} = \myvec{2&3&5\\ 1&2&4 \\ 7&6&2} - \myvec{2&3&5\\ 1&2&4 \\ 7&6&2}\\
    \implies 3\vec{A} - 5\vec{B} = \myvec{0&0&0\\ 0&0&0 \\ 0&0&0}
\end{align}
